# 2.SSRF жјҸжҙһеҺҹзҗҶпјҹ

ж„ҸжҖқжҳҜж”»еҮ»иҖ…е°ҶдёҖдёӘд»Јз ҒжҳҜдё»жңә A еҗ‘дё»жңә bпјҲеҶ…зҪ‘зҡ„дё»жңәпјүеҸ‘иө·иҜ·жұӮи®ҝй—®ж–Үд»¶

SSRF зҡ„еҪўжҲҗеӨ§еӨҡжҳҜз”ұдәҺжңҚеҠЎз«ҜжҸҗдҫӣдәҶд»Һе…¶д»–жңҚеҠЎеҷЁеә”з”ЁиҺ·еҸ–ж•°жҚ®зҡ„еҠҹиғҪдё”жІЎжңүеҜ№зӣ®ж Үең°еқҖеҒҡиҝҮж»ӨдёҺйҷҗеҲ¶гҖӮдҫӢеҰӮпјҢй»‘е®ўж“ҚдҪңжңҚеҠЎз«Ҝд»ҺжҢҮе®ҡ URL ең°еқҖиҺ·еҸ–зҪ‘йЎөж–Үжң¬еҶ…е®№пјҢеҠ иҪҪжҢҮе®ҡең°еқҖзҡ„еӣҫзүҮзӯүпјҢеҲ©з”Ёзҡ„жҳҜжңҚеҠЎз«Ҝзҡ„иҜ·жұӮдјӘйҖ гҖӮSSRF еҲ©з”ЁеӯҳеңЁзјәйҷ·зҡ„ Web еә”з”ЁдҪңдёәд»ЈзҗҶж”»еҮ»иҝңзЁӢе’Ңжң¬ең°зҡ„жңҚеҠЎеҷЁгҖӮ

# дё»иҰҒж”»еҮ»ж–№ејҸ

еҜ№еӨ–зҪ‘гҖҒжңҚеҠЎеҷЁжүҖеңЁеҶ…зҪ‘гҖҒжң¬ең°иҝӣиЎҢз«ҜеҸЈжү«жҸҸпјҢиҺ·еҸ–дёҖдәӣжңҚеҠЎзҡ„ banner дҝЎжҒҜгҖӮ

ж”»еҮ»иҝҗиЎҢеңЁеҶ…зҪ‘жҲ–жң¬ең°зҡ„еә”з”ЁзЁӢеәҸгҖӮ

еҜ№еҶ…зҪ‘ Web еә”з”ЁиҝӣиЎҢжҢҮзә№иҜҶеҲ«пјҢиҜҶеҲ«дјҒдёҡеҶ…йғЁзҡ„иө„дә§дҝЎжҒҜгҖӮ

ж”»еҮ»еҶ…еӨ–зҪ‘зҡ„ Web еә”з”ЁпјҢдё»иҰҒжҳҜдҪҝз”Ё HTTP GET иҜ·жұӮе°ұеҸҜд»Ҙе®һзҺ°зҡ„ж”»еҮ» (жҜ”еҰӮ struts2гҖҒSQli зӯү)гҖӮ

еҲ©з”Ё file еҚҸи®®иҜ»еҸ–жң¬ең°ж–Үд»¶зӯүгҖӮ

жјҸжҙһдә§з”ҹзӣёе…іеҮҪж•°пјҡ

пјҲ1пјүfile_get_contents()

file_get_content еҮҪж•°д»Һз”ЁжҲ·жҢҮе®ҡзҡ„ url иҺ·еҸ–еҶ…е®№пјҢз„¶еҗҺжҢҮе®ҡдёҖдёӘж–Үд»¶еҗҚ j иҝӣиЎҢдҝқеӯҳпјҢе№¶еұ•зӨәз»ҷз”ЁжҲ·гҖӮfile_put_content еҮҪж•°жҠҠдёҖдёӘеӯ—з¬ҰдёІеҶҷе…Ҙж–Үд»¶дёӯгҖӮ

пјҲ2пјүfsockopen()

<?php 
function GetFile($host,$port,$link) { 
    $fp = fsockopen($host, intval($port), $errno, $errstr, 30);   
    if (!$fp) { 
        echo "$errstr (error number $errno) \n"; 
    } else { 
        $out = "GET $link HTTP/1.1\r\n"; 
        $out .= "Host: $host\r\n"; 
        $out .= "Connection: Close\r\n\r\n"; 
        $out .= "\r\n"; 
        fwrite($fp, $out); 
        $contents=''; 
        while (!feof($fp)) { 
            $contents.= fgets($fp, 1024); 
        } 
        fclose($fp); 
        return $contents; 
    } 
}
?>

fsockopen еҮҪж•°е®һзҺ°еҜ№з”ЁжҲ·жҢҮе®ҡ url ж•°жҚ®зҡ„иҺ·еҸ–пјҢиҜҘеҮҪж•°дҪҝз”Ё socketпјҲз«ҜеҸЈпјүи·ҹжңҚеҠЎеҷЁе»әз«Ӣ tcp иҝһжҺҘпјҢдј иҫ“ж•°жҚ®гҖӮеҸҳйҮҸ host дёәдё»жңәеҗҚпјҢport дёәз«ҜеҸЈпјҢerrstr иЎЁзӨәй”ҷиҜҜдҝЎжҒҜе°Ҷд»Ҙеӯ—з¬ҰдёІзҡ„дҝЎжҒҜиҝ”еӣһпјҢ30 дёәж—¶йҷҗ

пјҲ3пјүcurl_exec()

<?php 
if (isset($_POST['url'])){
    $link = $_POST['url'];
    $curlobj = curl_init();// еҲӣе»әж–°зҡ„ cURL иө„жәҗ

    $result=curl_exec($curlobj);// жҠ“еҸ– URL е№¶жҠҠе®ғдј йҖ’з»ҷжөҸи§ҲеҷЁ
    curl_close($curlobj);// е…ій—ӯ cURL иө„жәҗпјҢе№¶дё”йҮҠж”ҫзі»з»ҹиө„жәҗ

    $filename = './curled/'.rand().'.txt';

}
?>

curl_exec еҮҪж•°з”ЁдәҺжү§иЎҢжҢҮе®ҡзҡ„ cURL дјҡиҜқ
